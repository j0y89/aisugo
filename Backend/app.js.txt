const express = require('express');
const bodyParser = require('body-parser');

const app = express();
const port = 3000;

app.use(bodyParser.json());
app.use(bodyParser.urlencoded({ extended: true }));

app.post('/generate-song', (req, res) => {
    const userInput = req.body.songInput;
    const selectedGenre = req.body.genre; // Ottieni il genere musicale inviato dal frontend
    const generatedSong = generateSong(userInput, selectedGenre); // Passa il genere musicale alla funzione di generazione della canzone
    res.json({ song: generatedSong });
});

function generateSong(input, genre) {
    // Simula la generazione della canzone
    // Puoi personalizzare ulteriormente la canzone in base al genere musicale
    return `Canzone generata dall'intelligenza artificiale basata sulla creazione di testi. Genere: ${genre}`;
}

app.listen(port, () => {
    console.log(`Server running at http://localhost:${port}`);
});
